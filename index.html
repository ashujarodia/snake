<canvas id="canvas" width="400" height="400"></canvas>
<script>
// Initialize canvas and context
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
// Set up the game board
const boardSize = 20;
const tileSize = canvas.width / boardSize;
const board = [];
for (let i = 0; i < boardSize; i++) {
board[i] = [];
for (let j = 0; j < boardSize; j++) {
board[i][j] = 0;
}
}
// Set up the snake
let snake = [
[10, 10],
[10, 11],
[10, 12],
];
let direction = "left";
// Set up the food
let food = [];
spawnFood();
// Set up the game loop
let score = 0;
let gameLoop = setInterval(() => {
// Move the snake
let head = getNextHead();
snake.unshift(head);
if (head[0] === food[0] && head[1] === food[1]) {
score++;
spawnFood();
} else {
snake.pop();
}
// Check for game over
if (isGameOver()) {
clearInterval(gameLoop);
alert(`Game over! Score: ${score}`);
}
// Clear the canvas
ctx.clearRect(0, 0, canvas.width, canvas.height);
// Draw the board
for (let i = 0; i < boardSize; i++) {
for (let j = 0; j < boardSize; j++) {
if (board[i][j] === 1) {
ctx.fillStyle = "black";
} else if (food[0] === i && food[1] === j) {
ctx.fillStyle = "red";
} else {
ctx.fillStyle = "white";
}
ctx.fillRect(i * tileSize, j * tileSize, tileSize, tileSize);
}
}
// Draw the snake
ctx.fillStyle = "green";
snake.forEach((segment) => {
ctx.fillRect(segment[0] * tileSize, segment[1] * tileSize, tileSize, tileSize);
});
// Update the direction
document.addEventListener("keydown", (event) => {
switch (event.key) {
case "ArrowUp":
if (direction !== "down") {
direction = "up";
}
break;
case "ArrowDown":
if (direction !== "up") {
direction = "down";
}
break;
case "ArrowLeft":
if (direction !== "right") {
direction = "left";
}
break;
case "ArrowRight":
if (direction !== "left") {
direction = "right";
}
break;
}
});
}, 100);
// Helper functions
function getNextHead() {
let head = snake[0].slice();
switch (direction) {
case "up":
head[1]--;
break;
case "down":
head[1]++;
break;
case "left":
head[0]--;
break;
case "right":
head[0]++;
break;
}
return head;
}
function spawnFood() {
do {
food[0] = Math.floor(Math.random() * boardSize);
food[1] = Math.floor(Math.random() * boardSize);
} while (board[food[0]][food[1]] === 1);
}
function isGameOver() {
// Check if snake hit a wall
if (snake[0][0] < 0 || snake[0][0] >= boardSize || snake[0][1] < 0 || snake[0][1] >=
boardSize) {
return true;
}
// Check if snake hit itself
for (let i = 1; i < snake.length; i++) {
if (snake[0][0] === snake[i][0] && snake[0][1] === snake[i][1]) {
return true;
}
}
return false;
}
</script>
